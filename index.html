function getSessionByTime(dateString) { // ቀኑ ካልተሰጠ ወደ 'London' ይመልሳል if (!dateString) return 'London'; const date = new Date(dateString); // የገባው ቀን ትክክለኛ መሆኑን ማረጋገጫ if (isNaN(date.getTime())) { return 'Invalid Date'; } // በUTC የቀኑን ቁጥር ማውጣት const day = date.getUTCDate(); // ክፍለ-ጊዜዎቹን በሊስት (Array) ማስቀመጥ (0=New York, 1=London, 2=Asian) const sessions = ['New York', 'London', 'Asian']; return sessions[day % 3]; } // ለሙከራ (Testing) console.log(getSessionByTime("2026-06-03")); // New York console.log(getSessionByTime("2026-06-04")); // London console.log(getSessionByTime("2026-06-05")); // Asian console.log(getSessionByTime("")); // London console.log(getSessionByTime("ስህተት ቀን")); // Invalid Date
